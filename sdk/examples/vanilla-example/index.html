Create your own custom theme with personalized colors and styling.

// Custom Theme Example
const customBooking = MovieBookingSDK.createIntegration({
  movieId: 'custom-movie',
  apiKey: 'demo-api-key',
  containerId: 'custom-widget',
  theme: {
    colors: {
      primary: '#8b5cf6',
      secondary: '#a855f7',
      background: '#1e1b4b',
      text: '#e0e7ff',
      accent: '#8b5cf6'
    },
    gradients: {
      primary: {
        type: 'linear',
        direction: '135deg',
        colors: ['#8b5cf6', '#7c3aed']
      }
    }
  }
});
